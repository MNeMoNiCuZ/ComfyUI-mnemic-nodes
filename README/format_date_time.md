```
%%Y = %Y = Year
%%m = %m = Month
%%d = %d = Day
%%H = %H = Hour
%%S = %S = Second
%%f = %f = Microsecond
%%x = %x = Date
%%X = %X = Time
%%c = %c = Date + Time
%%p = %p = am/pm
%%A = %A = Weekday
%%a = %a = Weekday short
%%B = %B = Month
%%b = %b = Month short
%%j = %j = Day of year
%%W = %W = Week index (Monday start)
%%w = %w = Weekday index (Monday start)
%%U = %U = Week index (Sunday start)
%%u = %u = Week index (Sunday start)
%%%% = %% = Prints a literal %%
```
